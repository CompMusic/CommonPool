%% III: Generic Musical concepts
%
\newglossaryentry{mode}{
  name=mode,
  description={Melodic framework. Synonymous to~\gls{makam},~\gls{raag},~\gls{raga} etc. in the most general sense.},
  sort=mode,
  type=MG
}

\newglossaryentry{tonic}{
  name=to\-nic,
  description={First degree of a scale. Synonymous to the~\gls{karar} in~\gls{OTMM} or~\Gls{sa} in~\gls{IAM}.},
  sort=tonic,
  type=MG
}

\newglossaryentry{cent}{
  name=cent,
  description={a logarithmic unit of measure used for musical intervals dividing an octave into $1200$ equidistant steps. A semitone in~$12$~\gls{TET} system is $100$ cents.},
  sort=cent,
  type=MG
}

\newglossaryentry{scale}{
  name=scale,
  description={A set of notes ordered with respect to a reference note},
  sort=scale,
  type=MG
}

\newglossaryentry{heterophony}{
  name=heterophony,
  description={simultaneous variation of a single melodic line XX},
  sort=heterophony,
  type=MG
}

\newacronym[firstplural=beats per minute (bpm), plural=bpm]{bpm}{bpm}{beat per minute}
\newacronym{TET}{TET}{tone-equal-tempered}
